import { Story, Preview } from '@storybook/addon-docs/blocks'; 
import { action } from '@storybook/addon-actions';
import { html } from 'lit-html';

import PfButton from './wrappers/button.wrapper.svelte';


# button (wrapped)

The good ol' button. This extends the native HTML button with a bit of styling...

<Meta title="atoms/button" />

### A simple button
<Preview withSource='open'>
    <Story name='default'>
        {{
            Component: PfButton,
            props: {
                content: '@propfull/button'
            },
            on: { click: action('on:click') }
        }}
    </Story>
</Preview>



### With icons or emojis
You can also add icons along with regular text.

<Preview withSource='open'>
    <Story name='with icon'>
        {{
            Component: PfButton,
            props: { content: 'download ⬇️' },
            on: { click: action('on:click') }
        }}
    </Story>
</Preview>

### Slot other elements
PfButtons can also have nested children as its content

<Preview>
    <Story name="nesting">
        {{
            Component: PfButton,
            props: {
                content: `
                <svg viewBox="0 0 8 8" width="5rem" height="5rem">
                    <circle cx="2" cy="2" r="1" fill="white" />
                    <circle cx="4" cy="4" r="1" fill="white" />
                    <circle cx="6" cy="6" r="1" fill="white" />
                </svg>
                <div>SVGs are awesome!</div>`
            },
            on: { click: action('on:click') }
        }}
    </Story>
</Preview>

### Theme
Buttons can also be themed - with the `theme` prop, the default being the `light` theme.
Here's how a button looks with the `dark` theme
<Preview withSource='open'>
    <Story name='dark'>
        {{
            Component: PfButton,
            props: {
                content: 'dark mode',
                theme: 'dark'
            },
            on: { click: action('on:click') }
        }}
    </Story>
</Preview>

### Other states
PfButtons can also have different states

#### 1. Disabled
This makes sure the button cannot be clicked and shows an appropriate cursor.

<Preview withSource='open'>
    <Story name='disabled'>
        {{
            Component: PfButton,
            props: {
                content: 'refreshing...',
                disabled: true
            },
            on: { click: action('on:click') }
        }}
    </Story>
</Preview>
